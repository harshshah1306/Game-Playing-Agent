import sys
data = sys.stdin.read().splitlines()
n=data[0]
datalist=[]
del data[0] #deletes the first element to store the rest in a list
for items in data:
    datalist.append(items.lower()) #converted to lower case as same alphabets could be both
s=sorted(datalist,key=len)  #sorted in ascending order of length
t=len(s[0])             #took the first length and stored in variable t for further use
count=0
visited={}   #to keep track for length of strings
for i in range(1,len(s)):
    if(len(s[i])==t):
        if(sorted(s[i])==sorted(s[i-1])):
            if not s[i] in visited:
                visited[s[i]]=1
            if not s[i-1] in visited:
                visited[s[i-1]]=1
    else:
        t=len(s[i])
print (len(visited))
